spacing and line breaks: pre

headlines: h1-h6
italic (visual): i
italic (meaning): em
bold (visual): b
bold (meaning): strong
lists: ul, ol
list elements: li
inline quote: q
block quote: blockquote
dates: date
subscripts, superscripts: sub, sup
small: small
markup for math: MathML
definition term: dt

  • image (only opening tag): img

audio:
  • audio (opening & closing tags): audio

video:
  • video: video

  • source attr: type

subtitles:
  • subtitles element within video element: track
  • attr: src, kind, label, srclang, kind
nothing (inline, only opening tag): span
nothing (block, only opening tag): div
